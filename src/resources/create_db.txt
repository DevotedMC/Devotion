CREATE TABLE IF NOT EXISTS dev_player_event (
    player_event_id BIGINT NOT NULL AUTO_INCREMENT
   ,trace_id VARCHAR(36) NOT NULL
   ,event_time DATETIME NOT NULL
   ,event_type VARCHAR(50) NOT NULL
   ,player_name VARCHAR(100) NOT NULL
   ,player_uuid VARCHAR(36)  NOT NULL
   ,location_worlduuid VARCHAR(36) NOT NULL
   ,location_x DOUBLE NOT NULL
   ,location_y DOUBLE NOT NULL
   ,location_z DOUBLE NOT NULL
   ,location_yaw FLOAT NOT NULL
   ,location_pitch FLOAT NOT NULL
   ,location_eye_worlduuid VARCHAR(36) NOT NULL
   ,location_eye_x DOUBLE NOT NULL
   ,location_eye_y DOUBLE NOT NULL
   ,location_eye_z DOUBLE NOT NULL
   ,location_eye_yaw FLOAT NOT NULL
   ,location_eye_pitch FLOAT NOT NULL
   ,game_mode VARCHAR(20)
   ,exhaustion FLOAT NOT NULL
   ,food_level INT NOT NULL
   ,saturation FLOAT NOT NULL
   ,total_experience INT NOT NULL
   ,velocity_x DOUBLE NOT NULL
   ,velocity_y DOUBLE NOT NULL
   ,velocity_z DOUBLE NOT NULL
   ,remaining_air INT NOT NULL
   ,health DOUBLE NOT NULL
   ,max_health DOUBLE NOT NULL
   ,status_flags CHAR(5) NOT NULL /* inVehicle, sneaking, sprinting, blocking, sleeping */
   
 --   ,CONSTRAINT pk_dev_player_event PRIMARY KEY (player_event_id)
);

CREATE TABLE IF NOT EXISTS dev_player_event_login (
    player_event_login_id BIGINT NOT NULL AUTO_INCREMENT
   ,trace_id VARCHAR(36) NOT NULL
   ,address VARCHAR(20)
   ,hostname VARCHAR(100)
   ,real_address VARCHAR(20)
   ,result VARCHAR(15)
   ,kick_message VARCHAR(100)
   
 --   ,CONSTRAINT pk_dev_player_event_login PRIMARY KEY (dev_player_event_login_id)
);

CREATE TABLE IF NOT EXISTS dev_player_event_interact (
    player_event_interact_id BIGINT NOT NULL AUTO_INCREMENT
   ,trace_id VARCHAR(36) NOT NULL
   ,item_type VARCHAR(50)
   ,item_amount INT
   ,item_durability SMALLINT
   ,item_enchantments VARCHAR(500)
   ,item_lore VARCHAR(500)
   ,action_name VARCHAR(20) NOT NULL
   ,clicked_block_type VARCHAR(50)
   ,block_face VARCHAR(20)
   ,event_cancelled BOOL NOT NULL
   
 --   ,CONSTRAINT pk_dev_player_event_interact PRIMARY KEY (player_event_interact_id)
);

CREATE TABLE IF NOT EXISTS dev_player_event_kick (
    player_event_kick_id BIGINT NOT NULL AUTO_INCREMENT
   ,trace_id VARCHAR(36) NOT NULL
   ,leave_message VARCHAR(500)
   ,kick_reason VARCHAR(500)
   
 --   ,CONSTRAINT pk_dev_player_event_kick PRIMARY KEY (player_event_kick_id)
);

CREATE TABLE IF NOT EXISTS dev_player_event_quit (
    dev_player_event_quit_id BIGINT NOT NULL AUTO_INCREMENT
   ,trace_id VARCHAR(36) NOT NULL
   ,quit_message VARCHAR(500)
   
 --   ,CONSTRAINT pk_dev_player_event_quit PRIMARY KEY (player_event_quit_id)
);

CREATE TABLE IF NOT EXISTS dev_player_event_teleport (
    player_event_teleport_id BIGINT NOT NULL AUTO_INCREMENT
   ,trace_id VARCHAR(36) NOT NULL
   ,cause VARCHAR(20)
   ,from_worlduuid VARCHAR(36) NOT NULL
   ,from_x DOUBLE NOT NULL
   ,from_y DOUBLE NOT NULL
   ,from_z DOUBLE NOT NULL
   ,from_yaw FLOAT NOT NULL
   ,from_pitch FLOAT NOT NULL
   ,to_worlduuid VARCHAR(36) NOT NULL
   ,to_x DOUBLE NOT NULL
   ,to_y DOUBLE NOT NULL
   ,to_z DOUBLE NOT NULL
   ,to_yaw FLOAT NOT NULL
   ,to_pitch FLOAT NOT NULL
   ,event_cancelled BOOL NOT NULL
      
 --   ,CONSTRAINT pk_dev_player_event_teleport PRIMARY KEY (player_event_teleport_id)
);

CREATE TABLE IF NOT EXISTS dev_player_event_respawn (
    player_event_respawn_id BIGINT NOT NULL AUTO_INCREMENT

 --   ,CONSTRAINT pk_dev_player_event_respawn PRIMARY KEY (player_event_respawn_id)
